    ```typescript
    import { readFileSync, existsSync } from "node:fs";
    import { join } from "node:path";
    import {
      shouldCompact,
      prepareCompaction,
      compact,
      DEFAULT_COMPACTION_SETTINGS,
      estimateTokens,
      type SessionEntry,
    } from "@mariozechner/pi-coding-agent";
    import { emitEvent } from "@emmy/telemetry";
    import type { ProfileSnapshot } from "@emmy/provider";
    import { markPreserved, type PreservationOpts } from "./preservation";
    import { loadCompactionConfig } from "./config-loader";
    import { SessionTooFullError, IllegalCompactionTimingError } from "./errors";

    export interface EmmyCompactionContext {
      profile: ProfileSnapshot;
      entries: SessionEntry[];
      contextTokens: number;
      contextWindow: number;          // profile.harness.context.max_input_tokens
      eventType: "turn_start" | string;
      model: unknown;                  // pi's Model reference
      apiKey: string;                  // pi's auth token reference
      setStatus?: (key: string, text: string | undefined) => void;  // optional ui.setStatus
    }

    export async function emmyCompactionTrigger(ctx: EmmyCompactionContext): Promise<{ ran: boolean; elided: number; preserved: number }> {
      // Pitfall #3 guard
      if (ctx.eventType !== "turn_start") {
        throw new IllegalCompactionTimingError(`emmyCompactionTrigger called on event=${ctx.eventType}; only turn_start is allowed (D-11 turn-boundary)`);
      }
      const cfg = loadCompactionConfig(ctx.profile);
      if (!cfg) return { ran: false, elided: 0, preserved: 0 };  // compaction disabled

      const ratio = ctx.contextTokens / ctx.contextWindow;
      if (ratio < cfg.soft_threshold_pct) return { ran: false, elided: 0, preserved: 0 };

      emitEvent({
        event: "session.compaction.trigger",
        ts: new Date().toISOString(),
        profile: ctx.profile.ref,
        reason: "soft_threshold",
        context_tokens: ctx.contextTokens,
        context_window: ctx.contextWindow,
        ratio,
      });

      // D-14 preservation pre-filter
      const preserveOpts: PreservationOpts = {
        structuralCore: true,
        errorPayloadsVerbatim: cfg.preserve_tool_results !== "none",
        activeGoal: true,
        recentTurns: cfg.preserve_recent_turns,
        filePins: true,
        todoState: true,
      };
      const preserved = markPreserved(ctx.entries, preserveOpts);
      const summarizable = ctx.entries.filter((e) => !preserved.has(e.uuid));

      // Hand pi only the summarizable subset
      const settings = { ...DEFAULT_COMPACTION_SETTINGS, enabled: true };
      const prep = prepareCompaction(summarizable, settings);
      if (!prep) return { ran: false, elided: 0, preserved: preserved.size };

      // D-17 visible status
      ctx.setStatus?.("emmy.compacting", `compacting ${prep.messagesToSummarize.length} turns…`);

      // D-13 profile-defined compaction prompt
      const promptFullPath = join(ctx.profile.ref.path, cfg.summarization_prompt_path);
      if (!existsSync(promptFullPath)) {
        // Treat as D-16 fallback instead of hard fail — session continues
        emitEvent({ event: "session.compaction.fallback", ts: new Date().toISOString(), profile: ctx.profile.ref, error: `compaction prompt missing at ${promptFullPath}` });
        return structuredPruneFallback(ctx, preserved);
      }
      const customInstructions = readFileSync(promptFullPath, "utf8");

      try {
        const result = await compact(prep, ctx.model, ctx.apiKey, undefined, customInstructions);
        ctx.setStatus?.("emmy.compacting", undefined);
        const summary_tokens = estimateTokens({ uuid: "_summary_", role: "user", content: [{ type: "text", text: result.summary }] } as SessionEntry);
        emitEvent({
          event: "session.compaction.complete",
          ts: new Date().toISOString(),
          profile: ctx.profile.ref,
          turns_elided: prep.messagesToSummarize.length,
          turns_preserved: preserved.size,
          summary_tokens,
        });
        // D-12 hard-ceiling guard after compaction
        const postTokens = ctx.contextTokens - prep.messagesToSummarize.reduce((s, e) => s + estimateTokens(e), 0) + summary_tokens;
        if (postTokens > ctx.contextWindow) {
          throw new SessionTooFullError({
            turn_index: ctx.entries.length,
            tokens: postTokens,
            max_input_tokens: ctx.contextWindow,
            compaction_attempt_result: { elided: prep.messagesToSummarize.length, summary_tokens },
            preservation_list: Array.from(preserved),
          });
        }
        return { ran: true, elided: prep.messagesToSummarize.length, preserved: preserved.size };
      } catch (err) {
        if (err instanceof SessionTooFullError) throw err;  // re-throw; D-12 fail-loud
        // D-16 fallback
        emitEvent({ event: "session.compaction.fallback", ts: new Date().toISOString(), profile: ctx.profile.ref, error: String(err) });
        return structuredPruneFallback(ctx, preserved);
      }
    }

    function structuredPruneFallback(ctx: EmmyCompactionContext, preserved: Set<string>): { ran: boolean; elided: number; preserved: number } {
      // Drop by age from non-preserved set until we're under 0.5 × window (aggressive)
      // This is a pure function modifying ctx.entries in place — but here we return elided count
      // Actual entry mutation happens via pi's session manager; emmy signals what to drop.
      const droppable = ctx.entries.filter((e) => !preserved.has(e.uuid)).sort((a, b) => {
        const ai = ctx.entries.indexOf(a); const bi = ctx.entries.indexOf(b);
        return ai - bi;  // oldest first
      });
      let toDrop = 0;
      let remaining = ctx.contextTokens;
      const target = ctx.contextWindow * 0.5;
      for (const e of droppable) {
        if (remaining <= target) break;
        remaining -= estimateTokens(e);
        toDrop++;
      }
      return { ran: true, elided: toDrop, preserved: preserved.size };
    }

    export { SessionTooFullError };
    ```

    import { createHash } from "node:crypto";

    export interface Sc2FixtureOpts {
      totalTurns: number;
      avgTurnTokens: number;
      errorTurnInterval: number;   // every Nth turn is error-flagged
      pinTurns: number[];          // indices where @file pins appear
      todoTurns: number[];         // indices where PLAN.md edits happen
    }

    export function generateSc2Fixture(opts: Sc2FixtureOpts): SessionEntry[] {
      const entries: SessionEntry[] = [];
      // Turn 0: system (structural core)
      entries.push({ uuid: "sys-0", role: "system", content: "EMMY ASSEMBLED (mock)\nprompt_sha256:deadbeef", isError: false });
      // Turn 1: goal user message (activeGoal)
      entries.push({ uuid: "u-1", role: "user", content: "implement full refactor across 40 files with tests" });
      // Turns 2..totalTurns-1: alternating assistant / tool_result, with error flags + pins + todos
      for (let i = 2; i < opts.totalTurns; i++) {
        const role = i % 2 === 0 ? "assistant" : "tool";
        const isError = role === "tool" && i % opts.errorTurnInterval === 0;
        let content: string;
        if (role === "assistant") {
          content = "x".repeat(opts.avgTurnTokens * 4);  // ~N chars ≈ N/4 tokens
        } else if (isError) {
          const stacktrace = (`Error: synthetic failure at /path:${i}\n  at fn${i} /path:${i}:col\n  at caller /path:${i-1}:col\n`).repeat(80);  // ~4KB
          content = stacktrace;
        } else {
          content = "ok" + "y".repeat(opts.avgTurnTokens * 4);
        }
        // File pins
        if (opts.pinTurns.includes(i)) {
          content = `@file:src/foo${i}.ts ${content}`;
        }
        // Todo edits
        if (opts.todoTurns.includes(i) && role === "tool") {
          content = `wrote PLAN.md: ${content}`;
        }
        entries.push({ uuid: `${role}-${i}`, role, content, isError });
      }
      return entries;
    }

    export function fixtureHash(fixture: SessionEntry[]): string {
      return createHash("sha256").update(JSON.stringify(fixture)).digest("hex");
    }
