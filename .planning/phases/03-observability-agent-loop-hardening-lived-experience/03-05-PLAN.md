    ```typescript
    export interface FeedbackRow {
      session_id: string; turn_id: string;
      profile_id: string; profile_version: string; profile_hash: string;
      rating: 1 | -1; comment: string;
      model_response: string; tool_calls: unknown[];
      latency_ms: number; kv_used: number;
      tokens_in: number; tokens_out: number;
    }

    const REQUIRED: readonly (keyof FeedbackRow)[] = [
      "session_id","turn_id","profile_id","profile_version","profile_hash",
      "rating","comment","model_response","tool_calls","latency_ms","kv_used","tokens_in","tokens_out",
    ];

    export class FeedbackSchemaError extends Error { constructor(public missing: string) { super(`feedback row missing required field: ${missing}`); } }
    export class FeedbackNotFoundError extends Error { constructor(public turnId: string) { super(`no feedback row with turn_id=${turnId}`); } }

    export function validateRow(r: Partial<FeedbackRow>): asserts r is FeedbackRow {
      for (const k of REQUIRED) {
        if (r[k] === undefined || r[k] === null) throw new FeedbackSchemaError(k);
      }
      if (r.rating !== 1 && r.rating !== -1) throw new FeedbackSchemaError("rating (must be 1 or -1)");
    }
    ```

    ```typescript
    import { existsSync, mkdirSync, readFileSync, renameSync, writeFileSync } from "node:fs";
    import { join, dirname } from "node:path";
    import { homedir } from "node:os";
    import { randomBytes } from "node:crypto";
    import { appendJsonlAtomic, writeJsonAtomic } from "./atomic-append";
    import { validateRow, FeedbackSchemaError, FeedbackNotFoundError, type FeedbackRow } from "./feedback-schema";

    export function defaultFeedbackPath(): string {
      return join(homedir(), ".emmy", "telemetry", "feedback.jsonl");
    }

    export function appendFeedback(path: string, row: FeedbackRow): void {
      validateRow(row);
      const serialized = JSON.stringify(row);
      if (serialized.length > 4096) {
        // large row — use tempfile+rename atomic write to a new line (reads existing, appends, rewrites whole file)
        const existing = existsSync(path) ? readFileSync(path, "utf8") : "";
        const newContent = existing + serialized + "\n";
        const dir = dirname(path);
        if (!existsSync(dir)) mkdirSync(dir, { recursive: true });
        const tmp = join(dir, `.${randomBytes(8).toString("hex")}.tmp`);
        writeFileSync(tmp, newContent, "utf8");
        renameSync(tmp, path);
      } else {
        appendJsonlAtomic(path, row as unknown as Record<string, unknown>);
      }
    }

    export function readFeedback(path: string): FeedbackRow[] {
      if (!existsSync(path)) return [];
      const text = readFileSync(path, "utf8");
      return text.split("\n").filter((l) => l.trim().length > 0).map((l) => JSON.parse(l) as FeedbackRow);
    }

    export function updateFeedback(path: string, turnId: string, patch: Partial<FeedbackRow>): void {
      const rows = readFeedback(path);
      const idx = rows.findIndex((r) => r.turn_id === turnId);
      if (idx < 0) throw new FeedbackNotFoundError(turnId);
      const updated = { ...rows[idx]!, ...patch };
      validateRow(updated);
      rows[idx] = updated;
      const newContent = rows.map((r) => JSON.stringify(r)).join("\n") + "\n";
      const dir = dirname(path);
      if (!existsSync(dir)) mkdirSync(dir, { recursive: true });
      const tmp = join(dir, `.${randomBytes(8).toString("hex")}.tmp`);
      writeFileSync(tmp, newContent, "utf8");
      renameSync(tmp, path);
    }

    export function upsertFeedback(path: string, row: FeedbackRow): void {
      // Idempotent helper for Alt+Up/Down flow: if turn_id exists, update; else append.
      validateRow(row);
      try {
        updateFeedback(path, row.turn_id, row);
      } catch (e) {
        if (e instanceof FeedbackNotFoundError) appendFeedback(path, row);
        else throw e;
      }
    }

    export { FeedbackSchemaError, FeedbackNotFoundError };
    ```

    import { existsSync, mkdirSync, readFileSync, copyFileSync, writeFileSync } from "node:fs";
    import { join } from "node:path";
    import { readFeedback } from "./feedback";

    export interface ExportResult {
      rowCount: number;
      warningCount: number;
      outDir: string;
    }

    const FILE_CONTENT_MARKERS = [/\n```/, /\nfunction\s+\w+\s*\(/, /\nimport\s+\w+\s+from\s+['"]/, /\nclass\s+\w+/, /\n#include\s+</];

    export function exportHfDataset(srcJsonl: string, outDir: string, opts: { emmyVersion?: string; gitSha?: string } = {}): ExportResult {
      if (!existsSync(srcJsonl)) throw new Error(`source feedback jsonl missing: ${srcJsonl}`);
      if (!existsSync(outDir)) mkdirSync(outDir, { recursive: true });

      // Copy JSONL verbatim (HF-loadable as-is per RESEARCH §Summary #5)
      copyFileSync(srcJsonl, join(outDir, "feedback.jsonl"));

      const rows = readFeedback(srcJsonl);
      let warnings = 0;
      for (const row of rows) {
        const blob = (row.model_response ?? "") + JSON.stringify(row.tool_calls);
        if (FILE_CONTENT_MARKERS.some((m) => m.test(blob))) {
          console.error(`[emmy/export-hf] WARNING: row turn_id=${row.turn_id} may contain file contents`);
          warnings++;
        }
      }

      // dataset_card.md
      const card = [
        `# Emmy Lived-Experience Feedback Dataset`,
        ``,
        `- **Rows:** ${rows.length}`,
        `- **Schema:** 13 fields (session_id, turn_id, profile_id, profile_version, profile_hash, rating ∈ {1,-1}, comment, model_response, tool_calls, latency_ms, kv_used, tokens_in, tokens_out)`,
        `- **Exported:** ${new Date().toISOString()}`,
        `- **Emmy version:** ${opts.emmyVersion ?? "unknown"}`,
        `- **Git SHA:** ${opts.gitSha ?? "unknown"}`,
        ``,
        `## Loading`,
        ``,
        "```python",
        `from datasets import load_dataset`,
        `d = load_dataset("json", data_files="${outDir}/feedback.jsonl")`,
        "```",
        ``,
        `## Warnings`,
        ``,
        `${warnings} row(s) may contain file contents (code blocks / function / import / class / #include markers).`,
        `Review before publication per TELEM-02 TODO(Phase-7) consent flow.`,
      ].join("\n");
      writeFileSync(join(outDir, "dataset_card.md"), card, "utf8");

      // provenance.json
      const provenance = {
        emmy_version: opts.emmyVersion ?? "unknown",
        git_sha: opts.gitSha ?? "unknown",
        export_ts: new Date().toISOString(),
        row_count: rows.length,
        warning_count: warnings,
        source_path: srcJsonl,
        profile_hashes: [...new Set(rows.map((r) => r.profile_hash))],
      };
      writeFileSync(join(outDir, "provenance.json"), JSON.stringify(provenance, null, 2), "utf8");

      return { rowCount: rows.length, warningCount: warnings, outDir };
    }

    ```typescript
      if (!ctx.enabled) return { action: "continue" };
      if (event.text !== ANSI_ALT_UP && event.text !== ANSI_ALT_DOWN) return { action: "continue" };

      const latest = tracker.getLatest();
      if (!latest) return { action: "continue" };  // no completed turn to rate

      const rating: 1 | -1 = event.text === ANSI_ALT_UP ? 1 : -1;

      let comment = "";
      if (rating === -1) {
        const typed = await ctx.ui.input("Thumbs-down — why?", "optional free-text (enter to skip)");
        comment = typed ?? "";
      }

      const row: FeedbackRow = {
        session_id: latest.session_id,
        turn_id: latest.turn_id,
        profile_id: latest.profile_id,
        profile_version: latest.profile_version,
        profile_hash: latest.profile_hash,
        rating,
        comment,
        model_response: latest.model_response,
        tool_calls: latest.tool_calls,
        latency_ms: latest.latency_ms,
        kv_used: latest.kv_used,
        tokens_in: latest.tokens_in,
        tokens_out: latest.tokens_out,
      };
      upsertFeedback(feedbackPath, row);  // idempotent — repeated press updates
      return { action: "handled" };
    }
    ```
