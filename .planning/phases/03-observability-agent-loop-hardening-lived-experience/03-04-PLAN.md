    ```typescript
    export interface MetricSample { ts: number; tokens: number }
    export interface MetricSnapshot { [name: string]: number }

    const METRIC_LINE = /^([a-zA-Z_:][a-zA-Z0-9_:]*)(?:\{[^}]*\})?\s+(.+?)\s*$/;

    export async function fetchVllmMetrics(baseUrl: string, timeoutMs = 2000): Promise<MetricSnapshot> {
      const url = `${baseUrl.replace(/\/$/, "")}/metrics`;
      const ctl = new AbortController();
      const t = setTimeout(() => ctl.abort(), timeoutMs);
      try {
        const resp = await fetch(url, { signal: ctl.signal });
        if (!resp.ok) throw new Error(`status ${resp.status}`);
        const text = await resp.text();
        return parseMetrics(text);
      } finally { clearTimeout(t); }
    }

    export function parseMetrics(text: string): MetricSnapshot {
      const out: MetricSnapshot = {};
      for (const line of text.split("\n")) {
        if (line.startsWith("#") || !line.trim()) continue;
        const m = METRIC_LINE.exec(line);
        if (m && m[1] && m[2]) {
          const v = parseFloat(m[2]);
          if (!Number.isNaN(v)) out[m[1]] = v;
        }
      }
      return out;
    }

    export class TokRateTracker {
      private _samples: MetricSample[] = [];
      private _windowMs = 5000;

      push(tokensTotal: number, nowMs: number = Date.now()): void {
        this._samples.push({ ts: nowMs, tokens: tokensTotal });
        this._samples = this._samples.filter((s) => nowMs - s.ts < this._windowMs);
      }
      rate(nowMs: number = Date.now()): number {
        this._samples = this._samples.filter((s) => nowMs - s.ts < this._windowMs);
        if (this._samples.length < 2) return 0;
        const first = this._samples[0]!;
        const last = this._samples[this._samples.length - 1]!;
        const dtSec = (last.ts - first.ts) / 1000;
        if (dtSec <= 0) return 0;
        return (last.tokens - first.tokens) / dtSec;
      }
      samplesInWindow(): number { return this._samples.length; }
    }

    export function computeTokRate(samples: MetricSample[]): number {
      if (samples.length < 2) return 0;
      const first = samples[0]!;
      const last = samples[samples.length - 1]!;
      const dtSec = (last.ts - first.ts) / 1000;
      if (dtSec <= 0) return 0;
      return (last.tokens - first.tokens) / dtSec;
    }
    ```

    ```typescript
    import { spawnSync } from "node:child_process";

    const NA_SENTINELS = new Set(["[n/a]", "n/a", "", "nan"]);

    export interface NvidiaSample {
      ts: string;
      gpu_util_pct?: number;
      gpu_clock_mhz?: number;
      gpu_temp_c?: number;
      memory_used_mb?: number;
    }

    export function parseFloatOrUndefined(raw: string): number | undefined {
      const s = (raw ?? "").trim();
      if (NA_SENTINELS.has(s.toLowerCase())) return undefined;
      const n = parseFloat(s);
      return Number.isNaN(n) ? undefined : n;
    }

    export function sampleNvidiaSmi(opts: { bin?: string; timeoutMs?: number } = {}): NvidiaSample | null {
      const bin = opts.bin ?? process.env.EMMY_NVIDIA_SMI_BIN ?? "nvidia-smi";
      const timeout = opts.timeoutMs ?? 5000;
      let out: ReturnType<typeof spawnSync>;
      try {
        out = spawnSync(bin, [
          "--query-gpu=timestamp,utilization.gpu,clocks.current.graphics,temperature.gpu,memory.used",
          "--format=csv,noheader,nounits",
        ], {
          encoding: "utf8",
          timeout,
          stdio: ["ignore", "pipe", "ignore"],
        });
      } catch { return null; }
      if (out.status !== 0 || !out.stdout) return null;
      const firstLine = out.stdout.split("\n").find((l) => l.trim().length > 0);
      if (!firstLine) return null;
      const parts = firstLine.split(",").map((p) => p.trim());
      if (parts.length < 5) return null;
      const [ts, util, clock, temp, mem] = parts;
      const sample: NvidiaSample = { ts: ts ?? "" };
      const u = parseFloatOrUndefined(util ?? ""); if (u !== undefined) sample.gpu_util_pct = u;
      const c = parseFloatOrUndefined(clock ?? ""); if (c !== undefined) sample.gpu_clock_mhz = c;
      const t = parseFloatOrUndefined(temp ?? ""); if (t !== undefined) sample.gpu_temp_c = t;
      const m = parseFloatOrUndefined(mem ?? ""); if (m !== undefined) sample.memory_used_mb = m;
      return sample;
    }
    ```

    ```typescript
    import { fetchVllmMetrics, TokRateTracker } from "./vllm-metrics";
    import { sampleNvidiaSmi } from "./nvidia-smi";
    import { formatFooter, type FooterValues } from "./footer";

    export interface FooterPollerHandle { stop: () => void }
    export interface FooterPollerOpts {
      baseUrl: string;                // vLLM endpoint (http://127.0.0.1:8002)
      setStatus: (key: string, text: string | undefined) => void;
      intervalMs?: number;            // default 1000 (D-23 = 1 Hz)
      maxFailures?: number;           // default 3 (D-24 degrade threshold)
      intervalImpl?: typeof setInterval;  // test hook
    }

    interface FieldState {
      lastValue: number | undefined;
      failCount: number;
    }

    export function startFooterPoller(opts: FooterPollerOpts): FooterPollerHandle {
      const intervalMs = opts.intervalMs ?? 1000;
      const maxFailures = opts.maxFailures ?? 3;
      const setInt = opts.intervalImpl ?? setInterval;
      const tokTracker = new TokRateTracker();
      const fields: Record<"gpu" | "kv" | "tok", FieldState> = {
        gpu: { lastValue: undefined, failCount: 0 },
        kv:  { lastValue: undefined, failCount: 0 },
        tok: { lastValue: undefined, failCount: 0 },
      };

      const tick = async (): Promise<void> => {
        const values: FooterValues = { specAccept: "-" };

        // vLLM /metrics
        try {
          const snap = await fetchVllmMetrics(opts.baseUrl, Math.max(500, intervalMs / 2));
          const kvPerc = snap["vllm:gpu_cache_usage_perc"];
          if (kvPerc !== undefined) {
            values.kvPct = kvPerc * 100;
            fields.kv.lastValue = values.kvPct;
            fields.kv.failCount = 0;
          } else {
            fields.kv.failCount++;
            if (fields.kv.failCount < maxFailures && fields.kv.lastValue !== undefined) {
              values.kvPct = fields.kv.lastValue;
            }
          }
          const totalTokens = snap["vllm:generation_tokens_total"];
          if (totalTokens !== undefined) {
            tokTracker.push(totalTokens);
            values.tokPerS = tokTracker.rate();
            fields.tok.lastValue = values.tokPerS;
            fields.tok.failCount = 0;
          } else {
            fields.tok.failCount++;
          }
        } catch {
          fields.kv.failCount++;
          fields.tok.failCount++;
          if (fields.kv.failCount < maxFailures && fields.kv.lastValue !== undefined) values.kvPct = fields.kv.lastValue;
          if (fields.tok.failCount < maxFailures && fields.tok.lastValue !== undefined) values.tokPerS = fields.tok.lastValue;
        }

        // nvidia-smi
        try {
          const smi = sampleNvidiaSmi();
          if (smi?.gpu_util_pct !== undefined) {
            values.gpuPct = smi.gpu_util_pct;
            fields.gpu.lastValue = values.gpuPct;
            fields.gpu.failCount = 0;
          } else {
            fields.gpu.failCount++;
            if (fields.gpu.failCount < maxFailures && fields.gpu.lastValue !== undefined) values.gpuPct = fields.gpu.lastValue;
          }
        } catch {
          fields.gpu.failCount++;
          if (fields.gpu.failCount < maxFailures && fields.gpu.lastValue !== undefined) values.gpuPct = fields.gpu.lastValue;
        }

        opts.setStatus("emmy.footer", formatFooter(values));
      };

      const handle = setInt(tick, intervalMs);
      // Prime immediately so footer doesn't show "--" for full interval
      void tick();

      return { stop: () => clearInterval(handle as ReturnType<typeof setInterval>) };
    }
    ```
