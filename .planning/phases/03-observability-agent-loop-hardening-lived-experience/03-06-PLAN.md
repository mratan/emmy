Design locks:
- **Audit compares tool host-set to `union(loopback_set, web_fetch.allowlist)`** (D-26). Not DNS-resolved (hostname-exact per RESEARCH Threat #6a).
- **Default allowlist is empty** → web_fetch itself flips badge red unless profile explicitly allowlists documentation hosts (`docs.python.org`, `developer.mozilla.org`, etc.).
- **Runtime enforcement at web_fetch call site** (D-27). Boot audit alone misses URL-level violations.

Output: Booting pi-emmy with 8 native tools + no MCP + no web_fetch allowlist → green OFFLINE OK (web_fetch is registered but declares no required hosts — see design note below). Adding `docs.python.org` to allowlist → still green. Calling `web_fetch("https://notes.example.com")` → badge flips red + session continues + violation logged.

**Design note on web_fetch host-set:** `web_fetch` itself declares `required_hosts: []` (no fixed host requirement — the URL is dynamic). The allowlist is the dynamic-host guardrail. This way:
- Tool-registry boot audit passes (web_fetch declares no specific hosts)
- Runtime allowlist check gates actual URLs at call time
- Without an allowlist, any `web_fetch(url)` flips badge red (default-deny)
</objective>

    **(1) `packages/emmy-telemetry/src/offline-audit.ts`** (~100 lines):
    ```typescript
    export const LOOPBACK_HOSTS: ReadonlySet<string> = new Set([
      "127.0.0.1", "localhost", "::1", "loopback", "0.0.0.0",
    ]);

    export interface EmmyToolRegistration {
      name: string;
      required_hosts: readonly string[];
    }

    export interface OfflineAuditResult {
      offline_ok: boolean;
      violating_tool: string | null;
      violating_host: string | null;
    }

    export function auditToolRegistry(
      tools: readonly EmmyToolRegistration[],
      webFetchAllowlist: readonly string[],
    ): OfflineAuditResult {
      const permitted = new Set<string>([...LOOPBACK_HOSTS, ...webFetchAllowlist]);
      for (const t of tools) {
        for (const h of t.required_hosts) {
          if (!permitted.has(h)) {
            return { offline_ok: false, violating_tool: t.name, violating_host: h };
          }
        }
      }
      return { offline_ok: true, violating_tool: null, violating_host: null };
    }

    export function auditWebFetchUrl(url: string, allowlist: readonly string[]): boolean {
      let hostname: string;
      try { hostname = new URL(url).hostname; }
      catch { throw new Error(`invalid URL: ${url}`); }
      if (LOOPBACK_HOSTS.has(hostname)) return true;  // loopback always ok
      // Hostname-exact; NO suffix matching (RESEARCH Threat T-03-06 CNAME bypass guard)
      return allowlist.includes(hostname);
    }
    ```

    import { auditWebFetchUrl } from "@emmy/telemetry";
    import { emitEvent } from "@emmy/telemetry";

    export class WebFetchAllowlistError extends Error {
      constructor(public url: string, public hostname: string) {
        super(`web_fetch blocked: host '${hostname}' not in allowlist (URL: ${url})`);
      }
    }

    export interface EnforcementContext {
      allowlist: readonly string[];
      profileRef: { id: string; version: string; hash: string };
      onViolation?: (details: { url: string; hostname: string }) => void;
    }

    export function enforceWebFetchAllowlist(url: string, ctx: EnforcementContext): void {
      if (!auditWebFetchUrl(url, ctx.allowlist)) {
        const hostname = new URL(url).hostname;
        emitEvent({
          event: "tool.web_fetch.violation",
          ts: new Date().toISOString(),
          profile: ctx.profileRef,
          url,
          hostname,
        });
        ctx.onViolation?.({ url, hostname });
        throw new WebFetchAllowlistError(url, hostname);
      }
    }
    ```

    import type { OfflineAuditResult } from "@emmy/telemetry";

    const GREEN = "\x1b[32m";
    const RED = "\x1b[31m";
    const BOLD = "\x1b[1m";
    const RESET = "\x1b[0m";

    export function renderBadge(result: OfflineAuditResult): string {
      if (result.offline_ok) return `${GREEN}${BOLD}OFFLINE OK${RESET}`;
      return `${RED}${BOLD}NETWORK USED${RESET} (${result.violating_tool ?? "?"} → ${result.violating_host ?? "?"})`;
    }

    export function renderBadgePlain(result: OfflineAuditResult): string {
      if (result.offline_ok) return "OFFLINE OK";
      return `NETWORK USED (${result.violating_tool ?? "?"} → ${result.violating_host ?? "?"})`;
    }

    export interface BadgeCtx { ui: { setStatus: (k: string, t: string | undefined) => void } }

    export function updateOfflineBadge(ctx: BadgeCtx, result: OfflineAuditResult): void {
      ctx.ui.setStatus("emmy.offline_badge", renderBadge(result));
    }

    // Module-level state for runtime badge flip on web_fetch violation
    let _lastResult: OfflineAuditResult = { offline_ok: true, violating_tool: null, violating_host: null };
    let _ctx: BadgeCtx | null = null;

    export function bindBadge(ctx: BadgeCtx): void {
      _ctx = ctx;
      updateOfflineBadge(ctx, _lastResult);
    }
    export function flipToViolation(tool: string, host: string): void {
      _lastResult = { offline_ok: false, violating_tool: tool, violating_host: host };
      if (_ctx) updateOfflineBadge(_ctx, _lastResult);
    }
    export function setInitialAudit(result: OfflineAuditResult): void {
      _lastResult = result;
      if (_ctx) updateOfflineBadge(_ctx, result);
    }
    ```
