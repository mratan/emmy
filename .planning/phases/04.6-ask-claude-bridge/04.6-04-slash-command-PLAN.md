```typescript
export interface AskClaudeResponse {
  response: string;
  duration_ms: number;
  rate_limit_remaining_hour: number;
}

export interface CallAskClaudeFn {
  (prompt: string): Promise<AskClaudeResponse>;
}

export function registerAskClaudeCommand(opts: {
  pi: PiCmd;
  callAskClaude: CallAskClaudeFn;
}): () => void {
  return opts.pi.registerCommand("ask-claude", {
    description: "Ask Claude (via Spark-side OAuth) a focused question — operator-invoked.",
    argHint: "<question>",
    handler: async (rawArg: string, ctx: SlashCmdCtx) => {
      const prompt = rawArg.trim();
      if (prompt.length === 0) {
        ctx.ui.notify("/ask-claude requires a question. Try: /ask-claude how do I…", "error");
        return;
      }
      try {
        const r = await opts.callAskClaude(prompt);
        ctx.ui.notify(
          `Claude (${r.duration_ms}ms, ${r.rate_limit_remaining_hour} calls left this hour):\n\n${r.response}`,
          "info",
        );
      } catch (err: any) {
        const reason: string = err?.reason ?? "unknown";
        const pattern: string | undefined = err?.pattern_class;
        switch (reason) {
          case "env_disabled":
            ctx.ui.notify(
              "/ask-claude is disabled on this Spark. Set EMMY_ASK_CLAUDE=on in the sidecar env to enable.",
              "error",
            );
            break;
          case "claude_cli_not_found":
            ctx.ui.notify(
              "/ask-claude: claude CLI not found on Spark PATH. Install Claude Code there first.",
              "error",
            );
            break;
          case "scrubber_blocked":
            ctx.ui.notify(
              `/ask-claude: prompt blocked by scrubber (${pattern ?? "secret pattern"}). Remove the matched content and retry.`,
              "error",
            );
            break;
          case "rate_limited_concurrent":
          case "rate_limited_min_gap":
          case "rate_limited_hourly":
            ctx.ui.notify(
              `/ask-claude: rate-limited (${reason}). Wait and retry.`,
              "error",
            );
            break;
          case "subprocess_failed":
          case "timeout":
            ctx.ui.notify(
              `/ask-claude: claude invocation failed (${reason}): ${err?.detail ?? ""}`,
              "error",
            );
            break;
          default:
            ctx.ui.notify(`/ask-claude: error — ${err?.message ?? "unknown"}`, "error");
        }
      }
    },
  });
}
```

```typescript
export interface AskClaudeError extends Error {
  reason: string;
  pattern_class?: string;
  detail?: string;
  exit_code?: number;
}

export async function callAskClaude(prompt: string): Promise<AskClaudeResponse> {
  const url = `${resolveEmmyServeBaseUrl()}/ask-claude`;
  const r = await fetch(url, {
    method: "POST",
    headers: { "content-type": "application/json" },
    body: JSON.stringify({ prompt }),
    // long timeout — Claude can take 30s+ for hard questions
    signal: AbortSignal.timeout(120_000),
  });
  if (!r.ok) {
    const body = await r.json().catch(() => ({}));
    const err = new Error(`ask-claude ${r.status}: ${body?.detail?.reason ?? body?.reason ?? r.statusText}`) as AskClaudeError;
    err.reason = body?.detail?.reason ?? body?.reason ?? "unknown";
    err.pattern_class = body?.detail?.pattern_class ?? body?.pattern_class;
    err.detail = body?.detail?.detail ?? body?.detail;
    throw err;
  }
  return await r.json();
}
```
