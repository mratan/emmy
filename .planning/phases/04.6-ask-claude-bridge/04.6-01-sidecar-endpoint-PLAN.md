### Task 2 — GREEN: implement the endpoint

Add to `emmy_serve/swap/controller.py`:

```python
# Pseudocode — fit real Pydantic + FastAPI patterns from existing controller

from emmy_serve.tools.scrubber import scrub  # from Plan 04.6-02

class AskClaudeRequest(BaseModel):
    prompt: str = Field(min_length=1, max_length=200_000)  # 200K char ceiling
    timeout_ms: int | None = Field(default=60_000, ge=1000, le=300_000)

class AskClaudeResponse(BaseModel):
    response: str
    duration_ms: int
    rate_limit_remaining_hour: int

class AskClaudeError(BaseModel):
    reason: str  # 'env_disabled' | 'claude_cli_not_found' | 'scrubber_blocked' | 'rate_limited_*' | 'subprocess_failed' | 'timeout'
    detail: str | None = None
    pattern_class: str | None = None  # only for scrubber_blocked

# In-memory rate-limit state (process-local; sidecar is single-process)
_ASK_CLAUDE_STATE = {
    "in_flight": False,
    "calls_this_hour": [],   # list of timestamps; prune older than 1h
    "last_call_ts": 0.0,
}
_ASK_CLAUDE_LOCK = asyncio.Lock()

@app.post("/ask-claude")
async def ask_claude(req: AskClaudeRequest) -> AskClaudeResponse:
    # 1. D-03 env gate
    if os.environ.get("EMMY_ASK_CLAUDE") != "on":
        raise HTTPException(503, detail={"reason": "env_disabled"})

    # 2. D-11 CLI gate
    if shutil.which("claude") is None:
        raise HTTPException(503, detail={"reason": "claude_cli_not_found"})

    # 3. D-06 scrubber
    scrub_result = scrub(req.prompt)
    if not scrub_result.clean:
        emit_event_ask_claude("scrubber_blocked", req.prompt, ...)
        raise HTTPException(400, detail={
            "reason": "scrubber_blocked",
            "pattern_class": scrub_result.matched_class,
        })

    # 4. D-07 rate-limit (under lock)
    async with _ASK_CLAUDE_LOCK:
        if _ASK_CLAUDE_STATE["in_flight"]:
            emit_event_ask_claude("rate_limited", req.prompt, reason="concurrent")
            raise HTTPException(429, detail={"reason": "rate_limited_concurrent"})
        now = time.monotonic()
        if now - _ASK_CLAUDE_STATE["last_call_ts"] < 3.0:
            emit_event_ask_claude("rate_limited", req.prompt, reason="min_gap")
            raise HTTPException(429, detail={"reason": "rate_limited_min_gap"})
        # prune > 1h
        cutoff = now - 3600
        _ASK_CLAUDE_STATE["calls_this_hour"] = [
            t for t in _ASK_CLAUDE_STATE["calls_this_hour"] if t > cutoff
        ]
        if len(_ASK_CLAUDE_STATE["calls_this_hour"]) >= 30:
            emit_event_ask_claude("rate_limited", req.prompt, reason="hourly")
            raise HTTPException(429, detail={"reason": "rate_limited_hourly"})
        _ASK_CLAUDE_STATE["in_flight"] = True
        _ASK_CLAUDE_STATE["last_call_ts"] = now
        _ASK_CLAUDE_STATE["calls_this_hour"].append(now)

    # 5. D-09 subprocess (argv + stdin, never shell)
    t0 = time.monotonic()
    emit_event_ask_claude("request", req.prompt, prompt_sha256=...)
    try:
        proc = await asyncio.create_subprocess_exec(
            "claude", "--print", "-",
            stdin=asyncio.subprocess.PIPE,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
        )
        stdout, stderr = await asyncio.wait_for(
            proc.communicate(input=req.prompt.encode("utf-8")),
            timeout=req.timeout_ms / 1000.0,
        )
        if proc.returncode != 0:
            emit_event_ask_claude("error", req.prompt, response=stderr.decode(),
                                   exit_code=proc.returncode)
            raise HTTPException(502, detail={
                "reason": "subprocess_failed",
                "detail": stderr.decode("utf-8", errors="replace")[:500],
                "exit_code": proc.returncode,
            })
        response_text = stdout.decode("utf-8", errors="replace")
        duration_ms = int((time.monotonic() - t0) * 1000)
        emit_event_ask_claude("response", req.prompt, response=response_text,
                               duration_ms=duration_ms)
        return AskClaudeResponse(
            response=response_text,
            duration_ms=duration_ms,
            rate_limit_remaining_hour=30 - len(_ASK_CLAUDE_STATE["calls_this_hour"]),
        )
    except asyncio.TimeoutError:
        emit_event_ask_claude("error", req.prompt, reason="timeout")
        try: proc.kill()
        except: pass
        raise HTTPException(504, detail={"reason": "timeout"})
    finally:
        async with _ASK_CLAUDE_LOCK:
            _ASK_CLAUDE_STATE["in_flight"] = False


def emit_event_ask_claude(kind: str, prompt: str, response: str = "",
                           duration_ms: int = 0, **extras) -> None:
    """D-08 — emit JSONL audit event with sha256s + lengths, NOT content.
    Full content only when EMMY_LOG_FULL=on."""
    event = {
        "ts": datetime.now(timezone.utc).isoformat(),
        "event": f"emmy.ask_claude.{kind}",
        "prompt_sha256": hashlib.sha256(prompt.encode("utf-8")).hexdigest(),
        "prompt_len": len(prompt),
        **extras,
    }
    if response:
        event["response_sha256"] = hashlib.sha256(response.encode("utf-8")).hexdigest()
        event["response_len"] = len(response)
    if duration_ms:
        event["duration_ms"] = duration_ms
    if os.environ.get("EMMY_LOG_FULL") == "on":
        event["prompt_full"] = prompt
        event["response_full"] = response
    append_jsonl_atomic(_audit_jsonl_path(), event)
```
