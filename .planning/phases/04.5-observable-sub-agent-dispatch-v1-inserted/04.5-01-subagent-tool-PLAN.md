<objective>
Ship the v1 `SubAgentTool` core: a `defineTool({ name: "Agent", ... })` factory that, given a parent's services + parentCwd + persona registry + model resolver, builds a Claude Code-style `Agent` tool the parent's model can call. Each call dispatches a child session per the persona's spawn pattern (A=lean shared services, B=per-persona services with **parentCwd** + persona AGENTS.md injected via `resourceLoaderOptions.agentsFilesOverride`), runs one turn, captures the child's final assistant text, and returns ONLY that text to the parent. Auto-compaction is OFF on the child; dispose() always fires in finally (including under abort). Tool allowlist passed via the `tools` option narrows pi's defaults per persona — V3 contract is pi-rejection-at-registration (the LOCKED behavior).

       ```typescript
       import { readFileSync } from "node:fs";
       import { resolve as pathResolve } from "node:path";
       import {
         createAgentSessionFromServices,
         createAgentSessionServices,
         SessionManager,
         type AgentSessionServices,
       } from "@mariozechner/pi-coding-agent";
       import type { CreateSubAgentToolOpts, SubAgentSpec } from "./types";
       import { runOneTurnReturningText } from "./run-one-turn";

       export async function dispatchSubAgent(
         opts: CreateSubAgentToolOpts,
         persona: SubAgentSpec,
         params: { description: string; prompt: string; model?: string },
       ): Promise<{ output: string; details: { persona: string; ok: boolean; pattern: "lean" | "persona" } }> {
         // Pattern A vs B branch — LOCKED in CONTEXT.md §decisions (REVISED 2026-04-26).
         // Pattern B: cwd is parentCwd (so tools operate on user's project); persona AGENTS.md
         // is injected via resourceLoaderOptions.agentsFilesOverride. This DECOUPLES persona
         // prompt source from tool cwd — the bug we caught in checker B2.
         let services: AgentSessionServices;
         if (persona.pattern === "persona") {
           if (!persona.personaDir) {
             throw new Error(`[Agent] persona "${persona.name}" pattern="persona" requires personaDir`);
           }
           const personaAgentsPath = pathResolve(persona.personaDir, "AGENTS.md");
           const personaAgentsContent =
             persona.agentsContent ?? readFileSync(personaAgentsPath, "utf8");
           services = await createAgentSessionServices({
             cwd: opts.parentCwd,                              // child operates on user's project
             authStorage: opts.parentServices.authStorage,     // SHARED — auth is profile-level
             resourceLoaderOptions: {
               // Inject persona AGENTS.md as an additional project-context file.
               // Pi reads parent's CLAUDE.md/AGENTS.md from cwd PLUS this injected content.
               agentsFilesOverride: (base) => ({
                 agentsFiles: [
                   ...base.agentsFiles,
                   { path: personaAgentsPath, content: personaAgentsContent },
                 ],
               }),
             } as any,
           });
         } else {
           services = opts.parentServices;                     // SHARED — Pattern A
         }

         const sessionManager = SessionManager.inMemory(opts.parentCwd);
         const modelId = params.model ?? persona.modelOverride ?? "default";
         if (params.model && persona.modelOverride && params.model !== persona.modelOverride) {
           console.warn(`[Agent] model override "${params.model}" requested but v1 is single-model; using parent's model`);
         }
         const model = opts.modelResolver(modelId);

         const { session: child } = await createAgentSessionFromServices({
           services,
           sessionManager,
           model: model as any,
           tools: persona.toolAllowlist,   // narrows pi's defaults — V3 pi-rejection-at-registration
         } as any);

         child.setAutoCompactionEnabled(false);

         try {
           const finalText = await runOneTurnReturningText(child, params.prompt);
           return {
             output: finalText,
             details: { persona: persona.name, ok: true, pattern: persona.pattern },
           };
         } finally {
           child.dispose();
         }
       }
       ```
