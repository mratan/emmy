Wire the OTel parent→tool→child span propagation contract into the Agent dispatcher (Plan 04.5-01 left a stub). Two helpers ship in `otel.ts`:

1. **`withAgentToolSpan`** — opens an explicit `agent.tool.Agent` span around the dispatcher call, owned by the Agent tool's `execute` body. This is the **W1 fix** — without this span, the trace tree is only 3 levels (parent_session → subagent.<name> → child_invoke), missing the per-tool-call span the LOCKED CONTEXT.md diagram specifies. 2. **`withSubagentSpan`** — opens a `subagent.<persona_name>` span parented to whatever is active. When called from inside `withAgentToolSpan`, the result is the LOCKED 4-level shape. The child's vLLM HTTP request inherits the trace context via AsyncLocalStorage (H5 confirmed) — no manual `traceparent` injection in v1; OTel HTTP auto-instrumentation in `@emmy/telemetry` handles the wire-level header.

       ```typescript
       import { context, trace, SpanStatusCode, type Span } from "@opentelemetry/api";

       export const SUBAGENT_TRACER_NAME = "emmy-subagent";

       // W1 — Level 2 of the LOCKED 4-level trace tree.
       // Wraps the Agent tool's execute body so the dispatcher's subagent span has an explicit
       // tool-call parent matching the CONTEXT.md §decisions trace-tree diagram (line 183).
       export async function withAgentToolSpan<T>(
         personaName: string,
         parentSessionId: string | undefined,
         inner: (span: Span) => Promise<T>,
       ): Promise<T> {
         const tracer = trace.getTracer(SUBAGENT_TRACER_NAME);
         return await tracer.startActiveSpan(
           "agent.tool.Agent",
           {
             attributes: {
               "gen_ai.tool.name": "Agent",
               "gen_ai.tool.persona": personaName,
               "gen_ai.conversation.id": parentSessionId ?? "<unknown>",
             },
           },
           async (span: Span) => {
             try {
               return await inner(span);
             } catch (err) {
               span.recordException(err as Error);
               span.setStatus({ code: SpanStatusCode.ERROR, message: (err as Error)?.message ?? String(err) });
               throw err;
             } finally {
               span.end();
             }
           },
         );
       }

       // Level 3 of the LOCKED 4-level trace tree.
       // Parented to `agent.tool.Agent` when called from inside withAgentToolSpan.
       export async function withSubagentSpan<T>(
         persona: { name: string; pattern: "lean" | "persona" },
         parentSessionId: string | undefined,
         inner: (span: Span) => Promise<T>,
       ): Promise<T> {
         const tracer = trace.getTracer(SUBAGENT_TRACER_NAME);
         const parentCtx = context.active();           // capture BEFORE startActiveSpan creates child
         const parentSpan = trace.getSpan(parentCtx);
         const parentSpanIdHex = parentSpan?.spanContext()?.spanId;

         return await tracer.startActiveSpan(
           `subagent.${persona.name}`,
           {
             attributes: {
               "gen_ai.agent.name": persona.name,
               "gen_ai.agent.id": persona.name,         // v1 fallback; profile-id stamp comes from EmmyProfileStampProcessor
               "gen_ai.conversation.id": parentSessionId ?? "<unknown>",
               "emmy.subagent.pattern": persona.pattern,
               ...(parentSpanIdHex ? { "emmy.subagent.parent_span_id": parentSpanIdHex } : {}),
             },
           },
           async (span: Span) => {
             try {
               // tracer.startActiveSpan already activates the new span as the active span.
               // OTel uses AsyncLocalStorage under the hood (H5 confirms this works through pi's tool dispatch).
               return await inner(span);
             } catch (err) {
               span.recordException(err as Error);
               span.setStatus({ code: SpanStatusCode.ERROR, message: (err as Error)?.message ?? String(err) });
               throw err;
             } finally {
               span.end();
             }
           },
         );
       }
       ```
