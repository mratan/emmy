    """Emit swap-progress events as one JSON line per phase to stdout.

    The four labels STOPPING/LOADING/WARMUP/READY are D-02 LOCKED verbatim and
    are consumed by packages/emmy-ux/src/profile-swap-runner.ts. Changing them

    """
    from __future__ import annotations
    import json, sys
    from datetime import datetime, timezone

    STOPPING: str = "stopping vLLM"
    LOADING: str = "loading weights"
    WARMUP: str = "warmup"
    READY: str = "ready"

    def _now_iso() -> str:
        return datetime.now(timezone.utc).isoformat()

    def emit(phase: str, pct: int | None = None) -> None:
        rec: dict[str, object] = {"ts": _now_iso(), "phase": phase}
        if pct is not None:
            rec["pct"] = pct
        # flush=True so the TS parent reads line-buffered (profile-swap-runner.ts)
        print(json.dumps(rec, sort_keys=False), flush=True)
    ```

    """Rollback by recursion into swap_profile with no_rollback=True.

    D-04: rollback goes through the SAME primitive, not a special case.
    """
    from __future__ import annotations
    import json, subprocess
    from pathlib import Path

    from .progress import emit

    def rollback(failed_new: Path, prior_old: Path, port: int, run_dir: Path) -> int:
        """Stop the failed engine and re-invoke swap_profile to restart the prior profile.

        Always prints a final `{"rolled_back": true, "rollback_succeeded": bool}` JSON
        line to stdout for the TS caller to parse.
        Returns 6 regardless (the envelope flag conveys the outcome).
        """
        from .orchestrator import swap_profile   # local import avoids circular
        emit("rollback: stopping failed engine")
        subprocess.run(["docker", "stop", "--time", "15", "emmy-serve"], check=False)
        subprocess.run(["docker", "rm", "emmy-serve"], check=False)
        emit("rollback: restarting prior profile")
        rc = swap_profile(failed_new, prior_old, port, run_dir, no_rollback=True)
        envelope = {"rolled_back": True, "rollback_succeeded": rc == 0}
        print(json.dumps(envelope), flush=True)
        return 6
    ```
