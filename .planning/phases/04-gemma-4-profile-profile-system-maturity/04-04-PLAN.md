    ```typescript
    import { readFileSync } from "node:fs";
    import yaml from "js-yaml";

    export class RoutesLoadError extends Error {}

    export interface RouteRef { profileId: string; variant: string; }
    export interface RoutesConfig {
        default: RouteRef;
        roles: Record<"plan" | "edit" | "critic", RouteRef>;
    }

    function parseRef(s: string, field: string): RouteRef {
        if (typeof s !== "string" || !s.includes("@")) throw new RoutesLoadError(`${field}: invalid ref shape: ${s}`);
        const [pid, variant] = s.split("@");
        if (!pid || !variant) throw new RoutesLoadError(`${field}: missing profile id or variant: ${s}`);
        return { profileId: pid, variant };
    }

    export function loadRoutes(path: string): RoutesConfig {
        const raw = yaml.load(readFileSync(path, "utf8")) as { default?: string; roles?: Record<string, string> } | null;
        if (!raw || typeof raw !== "object") throw new RoutesLoadError(`${path}: empty or non-object YAML`);
        const def = raw.default;
        if (typeof def !== "string") throw new RoutesLoadError(`${path}: missing 'default'`);
        const defaultRef = parseRef(def, "default");
        const rolesRaw = raw.roles ?? {};
        const roles: Record<string, RouteRef> = {};
        for (const key of ["plan", "edit", "critic"] as const) {
            const v = rolesRaw[key];
            // Roles default to `default` if absent
            roles[key] = typeof v === "string" ? parseRef(v, `roles.${key}`) : defaultRef;
        }
        return { default: defaultRef, roles: roles as RoutesConfig["roles"] };
    }
    ```
