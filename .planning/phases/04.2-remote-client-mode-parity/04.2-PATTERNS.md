```python
async def profile_swap(req: SwapRequest, request: Request) -> EventSourceResponse:
    # Transition guard: must be in READY (else 409).
    if state.state != SidecarState.READY:
        raise HTTPException(409, f"swap requires state=ready, currently={state.state.value}")

    async def event_generator():
        await state.transition_to(SidecarState.SWAPPING)

                from_path=req.from_, to_path=req.to, port=req.port
```

```python
from __future__ import annotations
import asyncio
from enum import Enum

class SidecarState(Enum):
    STOPPED = "stopped"
    STARTING = "starting"
    READY = "ready"
    SWAPPING = "swapping"
    DRAINING = "draining"
    ERROR = "error"

ALLOWED_TRANSITIONS: dict[SidecarState, set[SidecarState]] = {
    SidecarState.STOPPED:  {SidecarState.STARTING, SidecarState.ERROR},
    SidecarState.STARTING: {SidecarState.READY, SidecarState.ERROR},
    SidecarState.READY:    {SidecarState.SWAPPING, SidecarState.DRAINING, SidecarState.ERROR},
    SidecarState.SWAPPING: {SidecarState.READY, SidecarState.ERROR},
    SidecarState.DRAINING: {SidecarState.STOPPED, SidecarState.ERROR},
    SidecarState.ERROR:    {SidecarState.STARTING},
}

class InvalidTransitionError(Exception):
    def __init__(self, frm: SidecarState, to: SidecarState):
        super().__init__(f"invalid transition: {frm.value} -> {to.value}")
        self.frm = frm
        self.to = to

class StateMachine:
    def __init__(self) -> None:
        self._state = SidecarState.STOPPED
        self._lock = asyncio.Lock()

    async def transition_to(self, new: SidecarState) -> None:
        async with self._lock:
            if new not in ALLOWED_TRANSITIONS[self._state] and new != SidecarState.ERROR:
                raise InvalidTransitionError(self._state, new)
            self._state = new

    @property
    def state(self) -> SidecarState:
        return self._state
```

- Docstring at top with D-07 LOCKED transition table embedded for reviewers.
- Module-level constants (`ALLOWED_TRANSITIONS`) sibling to the class (mirror `progress.py` STOPPING/LOADING/WARMUP/READY).

- `asyncio.Lock` ‚Äî uvicorn runs handlers concurrently; preflight.py is sync.

**Imports + signature (lift verbatim from RESEARCH ¬ßKey signatures lines 603-638):**

```python
from __future__ import annotations
import asyncio
import json
from typing import AsyncIterator

async def run_orchestrator_subprocess(
    *,
    from_path: str | None,  # None for cold start (no --from arg)
    to_path: str,
    port: int = 8002,
    cwd: str | None = None,
) -> AsyncIterator[dict]:
    argv = ["uv", "run", "python", "-m", "emmy_serve.swap.orchestrator",
            "--to", to_path, "--port", str(port)]
    if from_path is not None:
        argv.extend(["--from", from_path])
    proc = await asyncio.create_subprocess_exec(
        *argv,
        stdout=asyncio.subprocess.PIPE,
        stderr=asyncio.subprocess.PIPE,
        cwd=cwd,
    )
    assert proc.stdout is not None
    async for line in proc.stdout:
        text = line.decode(errors="replace").strip()
        if not text:
            continue
        try:
            yield json.loads(text)
        except json.JSONDecodeError:
            continue  # silently ignore malformed (matches TS-side discipline)
    rc = await proc.wait()
    yield {"_internal_exit": rc}
```

**What to KEEP (this is critical ‚Äî locked across language boundary):**
- argv literal `["uv", "run", "python", "-m", "emmy_serve.swap.orchestrator", "--to", to_path, "--port", str(port)]` matches `profile-swap-runner.ts:83-95` byte-for-byte.
- The `try: json.loads(text); except JSONDecodeError: continue` swallow pattern ‚Äî same as `profile-swap-runner.ts:138-140` ("Non-JSON / malformed line ‚Äî ignore per contract").
- The `--from` arg is OMITTED when `from_path is None` so cold-start (D-02 idempotent path) doesn't pass an empty string.

**What to CHANGE (Python idioms vs Node):**
- `asyncio.create_subprocess_exec` (not `subprocess.Popen`) ‚Äî lets FastAPI's event loop interleave with the SSE stream's `request.is_disconnected()` polling.
- `AsyncIterator[dict]` return ‚Üí caller uses `async for rec in run_orchestrator_subprocess(...)`.
- Final `_internal_exit` sentinel record so caller routes exit codes uniformly.

```ts
export type FetchSseImpl = (
  url: string,
  init: RequestInit & { signal: AbortSignal },
) => Promise<Response>;

export async function runSwapAndStreamProgressHttp(args: {
  from: string;
  to: string;
  port: number;
  onProgress: (phase: string, pct?: number) => void;
  fetchSseImpl?: FetchSseImpl;

    signal: ctl.signal,

    return { exit: 1 };
```
