<objective>
Build the FastAPI sidecar that owns vLLM lifecycle on Spark — the always-on Python control plane the Mac client talks to over Tailscale. Implements D-01 graceful drain, D-02 idempotent /start with same-variant short-circuit + cross-variant swap reuse, D-03 poll-only /status, D-06 FastAPI+uvicorn+pydantic+SSE stack, D-07 LOCKED status payload schema, and C-06 SSE wrapping of the Phase-4 D-02 LOCKED JSON-per-line progress contract.

Purpose: This is the keystone of Phase 04.2 — without the sidecar, no remote-client mode is possible. By formalizing the harness↔emmy-serve hard boundary as HTTP+SSE (CLAUDE.md "Two hard boundaries; the rest soft" principle), the spawn-subprocess IPC becomes a wire protocol the Mac can speak.

Output: 4 new Python modules under emmy_serve/swap/ + extended __init__.py + 3 deps in pyproject.toml + 5 unit-test files. The sidecar process is bootable but NOT yet wired to systemd (Plan 02) and the TS dispatcher does NOT yet route to it (Plan 03).
</objective>

```python
from __future__ import annotations
import asyncio, subprocess, time
import httpx
from prometheus_client.parser import text_string_to_metric_families

_NA_SENTINELS = frozenset({"[n/a]", "n/a", "", "nan"})

# 1-second TTL caches per CONTEXT D-07 (subprocess fire-rate bounded).
_metrics_cache: tuple[float, dict[str, float]] | None = None
_temp_cache: tuple[float, float | None] | None = None

async def fetch_vllm_metrics(base_url: str, timeout_s: float = 2.0) -> dict[str, float]:
    """Async sibling of kv_finder/metrics.scrape_metrics."""
    async with httpx.AsyncClient(timeout=timeout_s) as client:
        r = await client.get(f"{base_url.rstrip('/')}/metrics")
        r.raise_for_status()
    out: dict[str, float] = {}
    for family in text_string_to_metric_families(r.text):
        for sample in family.samples:
            out[sample.name] = sample.value
    return out

async def fetch_vllm_metrics_cached(base_url: str = "http://127.0.0.1:8002") -> dict[str, float]:
    global _metrics_cache
    now = time.monotonic()
    if _metrics_cache is not None and (now - _metrics_cache[0]) < 1.0:
        return _metrics_cache[1]
    val = await fetch_vllm_metrics(base_url)
    _metrics_cache = (now, val)
    return val

def _parse_float_or_none(raw: str) -> float | None:
    s = (raw or "").strip()
    if s.casefold() in _NA_SENTINELS:
        return None
    try:
        return float(s)
    except ValueError:
        return None

def sample_gpu_temp() -> float | None:
    """Subprocess to nvidia-smi for temperature.gpu only."""
    try:
        r = subprocess.run(
            ["nvidia-smi", "--query-gpu=temperature.gpu", "--format=csv,noheader,nounits"],
            capture_output=True, text=True, timeout=3.0,
        )
        if r.returncode != 0:
            return None
        line = r.stdout.strip().split("\n")[0]
        return _parse_float_or_none(line)
    except (subprocess.TimeoutExpired, FileNotFoundError, OSError):
        return None

def sample_gpu_temp_cached() -> float | None:
    global _temp_cache
    now = time.monotonic()
    if _temp_cache is not None and (now - _temp_cache[0]) < 1.0:
        return _temp_cache[1]
    val = sample_gpu_temp()
    _temp_cache = (now, val)
    return val

# Test-only cache reset (mirror web-search.ts __resetSearchCountForTests pattern).
def __reset_caches_for_tests() -> None:
    global _metrics_cache, _temp_cache
    _metrics_cache = None
    _temp_cache = None
```
