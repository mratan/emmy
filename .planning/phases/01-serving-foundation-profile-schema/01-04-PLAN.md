**3. `emmy_serve/thermal/audit.py`** — implements RESEARCH.md §9.5 thresholds:

```python
from __future__ import annotations
from dataclasses import dataclass
from .corpus import ThermalPrompt, ALL_THERMAL_PROMPTS


@dataclass(frozen=True)
class AuditReport:
    total_prompts: int
    prefill_to_decode_ratio: float            # prefill_sum / decode_sum
    pct_prefill_gte_10k: float                 # fraction with expected_prefill_tokens ≥ 10000
    pct_includes_tool_call: float
    max_single_prompt_share: float             # max prompt's token-mass / total token-mass
    passes: bool
    failures: list[str]


def audit_corpus(prompts: list[ThermalPrompt] | None = None) -> AuditReport:
    if prompts is None:
        prompts = ALL_THERMAL_PROMPTS
    total = len(prompts)
    total_prefill = sum(p.expected_prefill_tokens for p in prompts)
    total_decode = sum(p.expected_decode_tokens for p in prompts)
    ratio = total_prefill / max(total_decode, 1)
    pct_10k = sum(1 for p in prompts if p.expected_prefill_tokens >= 10000) / total
    pct_tool = sum(1 for p in prompts if p.includes_tool_call) / total

    # max prompt share: sum of prefill+decode for a single prompt, as fraction of total
    total_tokens = total_prefill + total_decode
    max_share = max(
        (p.expected_prefill_tokens + p.expected_decode_tokens) / max(total_tokens, 1)
        for p in prompts
    )

    failures = []
    if not (0.5 <= ratio <= 2.0):
        failures.append(f"prefill:decode ratio {ratio:.2f} outside [1:2, 2:1] per §9.5")
    if pct_10k < 0.30:
        failures.append(f"only {pct_10k:.0%} of prompts have prefill >= 10K tokens (need ≥ 30%)")
    if pct_tool < 0.20:
        failures.append(f"only {pct_tool:.0%} of prompts include tool-call shapes (need ≥ 20%)")
    if max_share > 0.15:
        failures.append(f"single prompt share {max_share:.0%} > 15% cap")

    return AuditReport(
        total_prompts=total,
        prefill_to_decode_ratio=ratio,
        pct_prefill_gte_10k=pct_10k,
        pct_includes_tool_call=pct_tool,
        max_single_prompt_share=max_share,
        passes=(len(failures) == 0),
        failures=failures,
    )


def main(argv: list[str] | None = None) -> int:
    import argparse, sys, json
    ap = argparse.ArgumentParser()
    ap.add_argument('--format', choices=['text', 'json'], default='text')
    args = ap.parse_args(argv)
    report = audit_corpus()
    if args.format == 'json':
        from dataclasses import asdict
        print(json.dumps(asdict(report), indent=2))
    else:
        print(f"Total prompts: {report.total_prompts}")
        print(f"prefill:decode ratio: {report.prefill_to_decode_ratio:.2f}")
        print(f"% prefill ≥ 10K: {report.pct_prefill_gte_10k:.0%}")
        print(f"% tool-call shapes: {report.pct_includes_tool_call:.0%}")
        print(f"max single prompt share: {report.max_single_prompt_share:.0%}")
        print(f"PASSES: {report.passes}")
        for f in report.failures:
            print(f"  FAIL: {f}")
    return 0 if report.passes else 1


if __name__ == '__main__':
    import sys; sys.exit(main())
```

```python
from prometheus_client.parser import text_string_to_metric_families


def parse_preemption_metrics(text: str) -> dict[str, float]:
    """Extract the KV-relevant metrics. Unknown metrics → absent from dict."""
    out: dict[str, float] = {}
    for fam in text_string_to_metric_families(text):
```

```python
def _finder_subset() -> list[ThermalPrompt]:
    # Pick a mix: all 5 prior CODE_* (small prefill) + AGENT_10K_REFACTOR + AGENT_20K_MULTIFILE + AGENT_30K_HISTORY
    ids = {"code_01", "code_02", "code_03", "code_04", "code_05",

    return [p for p in ALL_THERMAL_PROMPTS if p.task_id in ids]


def drive_load(base_url: str, served_model_name: str, duration_s: int) -> dict:
    """Loop the finder subset until `duration_s` elapses. Returns summary stats."""
    subset = _finder_subset()
    t_start = time.monotonic()
    n_req, latencies, tokens_out = 0, [], 0
    while time.monotonic() - t_start < duration_s:
        for p in subset:
            if time.monotonic() - t_start >= duration_s:
                break
            t0 = time.monotonic()
            try:
                r = httpx.post(
                    f"{base_url}/v1/chat/completions",
                    json={
                        "model": served_model_name,
                        "messages": [{"role": "user", "content": p.prompt}],
                        "temperature": 0.2,
                        "max_tokens": min(p.max_tokens, p.expected_decode_tokens + 500),
                    },
                    timeout=300.0,
                )
                r.raise_for_status()
                data = r.json()
                tokens_out += data.get('usage', {}).get('completion_tokens', 0)
            except Exception:
                pass
            latencies.append(time.monotonic() - t0)
            n_req += 1
    latencies.sort()
    return {
        "n_requests": n_req,
        "tokens_generated": tokens_out,
        "p50_latency_ms": int(latencies[len(latencies)//2] * 1000) if latencies else 0,
        "p99_latency_ms": int(latencies[min(len(latencies)-1, int(len(latencies)*0.99))] * 1000) if latencies else 0,
        "duration_s": round(time.monotonic() - t_start, 1),
    }
```

```python
"""KV-budget bisection finder (D-13, SERVE-08). See RESEARCH.md §8."""
from __future__ import annotations
import argparse, json, subprocess, sys, time, yaml
from dataclasses import dataclass
from pathlib import Path
from ..diagnostics.atomic import write_json_atomic, append_jsonl_atomic, write_text_atomic
from ..diagnostics.layout import EmmyRunLayout, new_run_id
from ..profile.loader import load_serving, load_profile_manifest
from .metrics import scrape_metrics
from .load_driver import drive_load


@dataclass
class FinderState:
    current_value: float
    ok_value: float
    preempted_at: float | None
    bump_pct: float
    iters: int


def _check_dmesg_oom(since: str = "10 minutes ago") -> list[str]:
    try:
        out = subprocess.check_output(
            ["dmesg", "-T", "--since", since],
            stderr=subprocess.DEVNULL, timeout=30,
        ).decode(errors="replace")
        hits = [line for line in out.splitlines()
                if any(pat in line.lower() for pat in ("oom", "out of memory", "killed process"))]
        return hits
    except Exception:
        return []


def _rewrite_gpu_mem_util(serving_path: Path, value: float) -> None:
    raw = yaml.safe_load(serving_path.read_text())
    raw['engine']['gpu_memory_utilization'] = round(value, 3)
    write_text_atomic(serving_path, yaml.safe_dump(raw, sort_keys=False))


def run_finder(
    profile_path: Path,
    *,
    initial: float = 0.75,
    safety_margin_pct: float = 5.0,
    step_up_pct: float = 2.0,
    min_step_pct: float = 0.5,
    drive_minutes: int = 10,
    max_iters: int = 12,
    base_url: str = "http://127.0.0.1:8002",
    base_runs_dir: Path = Path("runs"),
) -> Path:
    """Run the bisection. Returns the run_dir path."""
    run_id = new_run_id()
    layout = EmmyRunLayout(base_dir=base_runs_dir, run_id=run_id, kind="kv-finder")
    layout.run_dir.mkdir(parents=True, exist_ok=True)

    serving_path = profile_path / "serving.yaml"
    manifest = load_profile_manifest(profile_path / "profile.yaml")
    served_model = load_serving(serving_path).engine.served_model_name

    state = FinderState(current_value=initial, ok_value=initial,
                        preempted_at=None, bump_pct=step_up_pct, iters=0)
    start_ts = time.time()

    while state.iters < max_iters:
        _rewrite_gpu_mem_util(serving_path, state.current_value)
        # DO NOT recompute profile.yaml.hash here — intermediate iterations; final write happens once.

        # Start vLLM via start_emmy.sh (it handles digest read + smoke canary)
        subprocess.run(["./scripts/start_emmy.sh", "--profile", str(profile_path)],
                       check=True, timeout=420)

        # Snapshot pre-drive metrics
        pre = scrape_metrics(base_url)
        load_stats = drive_load(base_url, served_model, drive_minutes * 60)
        post = scrape_metrics(base_url)

        preempt_delta = int(post.get('vllm:num_preemptions_total', 0) - pre.get('vllm:num_preemptions_total', 0))
        swapped_delta = int(post.get('vllm:num_requests_swapped', 0) - pre.get('vllm:num_requests_swapped', 0))
        dmesg_hits = _check_dmesg_oom()

        failure = 'none'
        if preempt_delta > 0 or swapped_delta > 0:
            failure = 'preemption'
        elif dmesg_hits:
            failure = 'oom'

        append_jsonl_atomic(layout.iterations_path, {
            'iter': state.iters, 'ts': time.strftime('%Y-%m-%dT%H:%M:%SZ', time.gmtime()),
            'value': state.current_value, 'failure': failure,
            'duration_s': load_stats['duration_s'],
            'metrics': {
                'preemptions_delta': preempt_delta,
                'swapped_delta': swapped_delta,
                'p50_latency_ms': load_stats['p50_latency_ms'],
                'p99_latency_ms': load_stats['p99_latency_ms'],
                'tokens_generated': load_stats['tokens_generated'],
                'n_requests': load_stats['n_requests'],
            },
            'dmesg_matches': dmesg_hits,
        })

        subprocess.run(["docker", "stop", "emmy-serve"], check=False, timeout=60)
        subprocess.run(["docker", "rm", "emmy-serve"], check=False, timeout=30)

        # Update state
        if failure == 'none':
            state.ok_value = max(state.ok_value, state.current_value)
            if state.preempted_at is not None:
                next_val = (state.ok_value + state.preempted_at) / 2
                state.bump_pct /= 2
            else:
                next_val = state.current_value + state.bump_pct / 100
        else:
            state.preempted_at = min(state.preempted_at or 1.0, state.current_value)
            next_val = (state.ok_value + state.preempted_at) / 2
            state.bump_pct /= 2

        if abs(next_val - state.current_value) < min_step_pct / 100:
            break
        if next_val >= 1.0:
            break
        state.current_value = round(next_val, 3)
        state.iters += 1

    # Apply safety margin
    final_val = round(max(0.50, state.ok_value - safety_margin_pct / 100), 3)
    _rewrite_gpu_mem_util(serving_path, final_val)

    # Recompute profile hash once, after final write
    subprocess.run(["uv", "run", "emmy", "profile", "hash", str(profile_path), "--write"],
                   check=True, timeout=30)

    # Append measured-values block to PROFILE_NOTES.md
    _append_profile_notes(profile_path, final_val, state, run_id)

    summary = {
        'profile_id': manifest.profile.id,
        'profile_version': manifest.profile.version,
        'hardware_id': _hardware_id(),
        'initial_value': initial,
        'final_value': final_val,
        'safety_margin_pct': safety_margin_pct,
        'first_preemption_at': state.preempted_at,
        'highest_clean_value': state.ok_value,
        'iterations': state.iters,
        'total_duration_s': round(time.time() - start_ts, 1),
        'load_driver': 'kv_finder_subset_v1',
        'started': time.strftime('%Y-%m-%dT%H:%M:%SZ', time.gmtime(start_ts)),
        'finished': time.strftime('%Y-%m-%dT%H:%M:%SZ', time.gmtime()),
    }
    write_json_atomic(layout.summary_path, summary)
    print(f"Finder complete: final gpu_memory_utilization = {final_val}")
    print(f"  runs: {layout.run_dir}")
    return layout.run_dir


def _hardware_id() -> str:
    try:
        return subprocess.check_output(["hostname"], text=True).strip()
    except Exception:
        return "unknown"


def _append_profile_notes(profile_path: Path, final_val: float, state: FinderState, run_id: str):
    notes = profile_path / "PROFILE_NOTES.md"
    block = f"\n### KV-finder result (run {run_id})\n\n" \
            f"| Measurement | Value |\n|---|---|\n" \
            f"| `gpu_memory_utilization` (final) | {final_val} |\n" \
            f"| First-preemption value | {state.preempted_at} |\n" \
            f"| Highest clean value | {state.ok_value} |\n" \
            f"| Iterations | {state.iters} |\n" \
            f"| Hardware | {_hardware_id()} |\n" \
            f"| Run artifact | `runs/{run_id}-kv-finder/` |\n"
    with open(notes, 'a') as f:
        f.write(block)


def main(argv: list[str] | None = None) -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument('--profile', required=True)
    ap.add_argument('--drive-minutes', type=int, default=10)
    ap.add_argument('--max-iters', type=int, default=12)
    ap.add_argument('--initial', type=float, default=0.75)
    ap.add_argument('--safety-margin-pct', type=float, default=5.0)
    args = ap.parse_args(argv)
    run_finder(
        Path(args.profile),
        initial=args.initial,
        safety_margin_pct=args.safety_margin_pct,
        drive_minutes=args.drive_minutes,
        max_iters=args.max_iters,
    )
    return 0


if __name__ == '__main__':
    sys.exit(main())
```

```python
"""GPU + vLLM metrics samplers. Runs in background threads during thermal replay."""
from __future__ import annotations
import subprocess, threading, time
from pathlib import Path
from ..diagnostics.atomic import append_jsonl_atomic
from ..kv_finder.metrics import scrape_metrics


class GpuSampler(threading.Thread):
    """Polls nvidia-smi every `interval_s` seconds. Appends to JSONL."""
    def __init__(self, jsonl_path: Path, interval_s: float = 5.0, t_start: float = None):
        super().__init__(daemon=True)
        self.jsonl_path = jsonl_path
        self.interval_s = interval_s
        self.stop_evt = threading.Event()
        self.t_start = t_start or time.monotonic()

    def run(self):
        while not self.stop_evt.is_set():
            sample = self._sample()
            if sample:
                sample["t_elapsed"] = time.monotonic() - self.t_start
                append_jsonl_atomic(self.jsonl_path, sample)
            self.stop_evt.wait(self.interval_s)

    def stop(self):
        self.stop_evt.set()

    @staticmethod
    def _sample() -> dict | None:
        try:
            out = subprocess.check_output(
                ["nvidia-smi", "--query-gpu=timestamp,utilization.gpu,clocks.current.graphics,temperature.gpu,memory.used",
                 "--format=csv,noheader,nounits"],
                timeout=5, text=True,
            )
            line = out.strip().splitlines()[0]
            ts, util, clock, temp, mem = [p.strip() for p in line.split(",")]
            return {
                "ts": ts,
                "gpu_util_pct": float(util),
                "gpu_clock_mhz": float(clock),
                "gpu_temp_c": float(temp),
                "memory_used_mb": float(mem),
            }
        except Exception:
            return None


class VllmMetricsSampler(threading.Thread):
    def __init__(self, jsonl_path: Path, base_url: str, interval_s: float = 5.0, t_start: float = None):
        super().__init__(daemon=True)
        self.jsonl_path = jsonl_path
        self.base_url = base_url
        self.interval_s = interval_s
        self.stop_evt = threading.Event()
        self.t_start = t_start or time.monotonic()

    def run(self):
        while not self.stop_evt.is_set():
            try:
                m = scrape_metrics(self.base_url)
                m["t_elapsed"] = time.monotonic() - self.t_start
                m["ts"] = time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())
                append_jsonl_atomic(self.jsonl_path, m)
            except Exception:
                pass
            self.stop_evt.wait(self.interval_s)

    def stop(self):
        self.stop_evt.set()
```

```python
"""2-hour replay harness per RESEARCH.md §9.6 + §9.7."""
from __future__ import annotations
import argparse, itertools, json, subprocess, sys, time, yaml
from pathlib import Path
import httpx
from ..diagnostics.atomic import append_jsonl_atomic, write_json_atomic, write_text_atomic

from ..profile.loader import load_profile
from ..canary import run_sp_ok, run_tool_call, run_generate, load_default_tool_schema
from .corpus import ALL_THERMAL_PROMPTS
from .sampler import GpuSampler, VllmMetricsSampler


def _percentile(sorted_list, p: float) -> float:
    if not sorted_list:
        return 0.0
    idx = max(0, min(len(sorted_list) - 1, int(len(sorted_list) * p / 100)))
    return float(sorted_list[idx])


def run_replay(
    profile_path: Path,
    *,
    target_wall_time_s: int = 7200,
    inter_request_gap_s: float = 5.0,
    sample_interval_s: float = 5.0,
    base_url: str = "http://127.0.0.1:8002",
    base_runs_dir: Path = Path("runs"),
) -> Path:
    serving, _, profile_ref = load_profile(profile_path)
    served_model = serving.engine.served_model_name

    run_id = new_run_id()
    layout = EmmyRunLayout(base_dir=base_runs_dir, run_id=run_id, kind="thermal")
    layout.run_dir.mkdir(parents=True, exist_ok=True)

    # --- Pre-replay canary gate (§9.6) ---
    ok, _ = run_sp_ok(base_url, served_model)
    assert ok, "pre-replay SP_OK canary failed; abort"
    ok, _ = run_tool_call(base_url, served_model, load_default_tool_schema())
    assert ok, "pre-replay tool_call canary failed; abort"
    ok, _, _ = run_generate(base_url, served_model)
    assert ok, "pre-replay generate canary failed; abort"

    t_start = time.monotonic()
    gpu_sampler = GpuSampler(layout.gpu_samples_path, sample_interval_s, t_start=t_start)
    vllm_sampler = VllmMetricsSampler(layout.vllm_metrics_path, base_url, sample_interval_s, t_start=t_start)
    gpu_sampler.start()
    vllm_sampler.start()

    try:
        corpus_iter = itertools.cycle(ALL_THERMAL_PROMPTS)
        while time.monotonic() - t_start < target_wall_time_s:
            prompt = next(corpus_iter)
            t_req_start = time.monotonic()
            try:
                r = httpx.post(
                    f"{base_url}/v1/chat/completions",
                    json={
                        "model": served_model,
                        "messages": [{"role": "user", "content": prompt.prompt}],
                        "temperature": 0.2,
                        "max_tokens": prompt.max_tokens,
                    },
                    timeout=600.0,
                )
                r.raise_for_status()
                data = r.json()
                n_tokens = data.get("usage", {}).get("completion_tokens", 0)
                t_req_end = time.monotonic()
                duration_s = t_req_end - t_req_start
                append_jsonl_atomic(layout.responses_path, {
                    "task_id": prompt.task_id,
                    "t_start": t_req_start - t_start,
                    "duration_s": round(duration_s, 3),
                    "tokens_out": n_tokens,
                    "tokens_per_second": round(n_tokens / duration_s, 2) if duration_s > 0 else 0,
                })
                append_jsonl_atomic(layout.prompts_used_path, {
                    "t_elapsed": t_req_start - t_start,
                    "task_id": prompt.task_id,
                    "category": prompt.category,
                })
            except Exception as e:
                append_jsonl_atomic(layout.responses_path, {
                    "task_id": prompt.task_id,
                    "t_start": time.monotonic() - t_start,
                    "error": str(e),
                })
            time.sleep(inter_request_gap_s)
    finally:
        gpu_sampler.stop()
        vllm_sampler.stop()
        gpu_sampler.join(timeout=10)
        vllm_sampler.join(timeout=10)

    # dmesg tail
    try:
        out = subprocess.check_output(["dmesg", "-T", "--since", "2 hours ago"], timeout=30).decode(errors="replace")
        write_text_atomic(layout.dmesg_tail_path, out)
    except Exception:
        write_text_atomic(layout.dmesg_tail_path, "(dmesg unavailable)\n")

    summary = compute_floors(layout.run_dir)
    summary["profile_id"] = profile_ref.id
    summary["profile_version"] = profile_ref.version
    summary["target_wall_time_s"] = target_wall_time_s
    write_json_atomic(layout.summary_path, summary)
    return layout.run_dir


def _load_jsonl(path: Path) -> list[dict]:
    if not path.exists():
        return []
    return [json.loads(l) for l in path.read_text().splitlines() if l.strip()]


def compute_floors(run_dir: Path) -> dict:
    """Per RESEARCH.md §9.7."""
    gpu = [s for s in _load_jsonl(run_dir / "gpu_samples.jsonl") if s.get("t_elapsed", 0) >= 3600]
    reqs = [r for r in _load_jsonl(run_dir / "responses.jsonl") if r.get("t_start", 0) >= 3600]
    metrics = _load_jsonl(run_dir / "vllm_metrics.jsonl")

    gpu_clocks = sorted(s["gpu_clock_mhz"] for s in gpu if "gpu_clock_mhz" in s)
    gpu_temps = sorted(s["gpu_temp_c"] for s in gpu if "gpu_temp_c" in s)
    throughputs = sorted(r["tokens_per_second"] for r in reqs if "tokens_per_second" in r and r.get("tokens_per_second", 0) > 0)

    # preemptions over hour 2: last minus first vllm metric
    h2_metrics = [m for m in metrics if m.get("t_elapsed", 0) >= 3600]
    preempt_h2 = 0
    if h2_metrics:
        preempt_h2 = int(h2_metrics[-1].get("vllm:num_preemptions_total", 0) - h2_metrics[0].get("vllm:num_preemptions_total", 0))

    oom = 0
    dmesg = run_dir / "dmesg_tail.txt"
    if dmesg.exists():
        text = dmesg.read_text().lower()
        oom = sum(1 for line in text.splitlines() if "oom" in line or "out of memory" in line)

    return {
        "gpu_clock_p5_hour2_mhz": _percentile(gpu_clocks, 5),
        "gpu_clock_p50_hour2_mhz": _percentile(gpu_clocks, 50),
        "gpu_temp_p95_hour2_c": _percentile(gpu_temps, 95),
        "decode_throughput_p50_hour2_tokps": _percentile(throughputs, 50),
        "decode_throughput_p1_hour2_tokps": _percentile(throughputs, 1),
        "preemptions_hour2": preempt_h2,
        "oom_events": oom,
    }


def assert_floors(run_dir: Path, profile_path: Path) -> int:
    """Compare computed floors to PROFILE_NOTES.md recorded floors. Exit 0 if all pass."""
    computed = compute_floors(run_dir)
    notes = (profile_path / "PROFILE_NOTES.md").read_text()
    # Parse frontmatter YAML from top of PROFILE_NOTES.md
    if not notes.startswith("---"):
        print("PROFILE_NOTES.md has no YAML frontmatter; cannot assert floors", file=sys.stderr)
        return 1
    end = notes.find("\n---", 3)
    fm = yaml.safe_load(notes[3:end])
    recorded = fm.get("measured_values", {})

    failures = []
    # hard gates
    if computed["preemptions_hour2"] != 0:
        failures.append(f"preemptions_hour2={computed['preemptions_hour2']} (hard gate: must be 0)")
    if computed["oom_events"] != 0:
        failures.append(f"oom_events={computed['oom_events']} (hard gate: must be 0)")
    # tolerance gates
    r_clock = recorded.get("gpu_clock_p5_hour2_mhz")
    if r_clock and computed["gpu_clock_p5_hour2_mhz"] < r_clock * 0.95:
        failures.append(f"gpu_clock_p5 {computed['gpu_clock_p5_hour2_mhz']:.0f} < recorded {r_clock:.0f} * 0.95")
    r_p50 = recorded.get("decode_throughput_p50_hour2_tokps")
    if r_p50 and computed["decode_throughput_p50_hour2_tokps"] < r_p50 * 0.93:
        failures.append(f"decode_p50 {computed['decode_throughput_p50_hour2_tokps']:.1f} < recorded {r_p50:.1f} * 0.93")
    r_p1 = recorded.get("decode_throughput_p1_hour2_tokps")
    if r_p1 and computed["decode_throughput_p1_hour2_tokps"] < r_p1 * 0.90:
        failures.append(f"decode_p1 {computed['decode_throughput_p1_hour2_tokps']:.1f} < recorded {r_p1:.1f} * 0.90")

    if failures:
        for f in failures:
            print(f"FLOOR FAIL: {f}", file=sys.stderr)
        return 1
    print("All floors pass", file=sys.stderr)
    return 0


def record_floors_first_run(run_dir: Path, profile_path: Path) -> None:
    """First-run helper: writes computed floors into PROFILE_NOTES.md measured_values."""
    computed = compute_floors(run_dir)
    notes_path = profile_path / "PROFILE_NOTES.md"
    text = notes_path.read_text()
    assert text.startswith("---"), "expected YAML frontmatter"
    end = text.find("\n---", 3)
    fm = yaml.safe_load(text[3:end]) or {}
    mv = fm.setdefault("measured_values", {})
    mv["gpu_clock_p5_hour2_mhz"] = round(computed["gpu_clock_p5_hour2_mhz"], 0)
    mv["decode_throughput_p50_hour2_tokps"] = round(computed["decode_throughput_p50_hour2_tokps"], 1)
    mv["decode_throughput_p1_hour2_tokps"] = round(computed["decode_throughput_p1_hour2_tokps"], 1)
    # Rebuild file with new frontmatter
    new_fm = yaml.safe_dump(fm, sort_keys=False).strip()
    body = text[end + len("\n---"):]
    write_text_atomic(notes_path, f"---\n{new_fm}\n---{body}")
    # Recompute profile hash after measured values written
    subprocess.run(["uv", "run", "emmy", "profile", "hash", str(profile_path), "--write"], check=True)


def main(argv: list[str] | None = None) -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument('--profile', required=True)
    ap.add_argument('--target-wall-time-s', type=int, default=7200)
    ap.add_argument('--assert-floors', action='store_true',
                    help='compare measured floors to PROFILE_NOTES.md recorded values; fail if violated')
    ap.add_argument('--record-floors', action='store_true',
                    help='first-run: write computed floors into PROFILE_NOTES.md')
    args = ap.parse_args(argv)
    run_dir = run_replay(Path(args.profile), target_wall_time_s=args.target_wall_time_s)
    print(f"thermal replay complete: {run_dir}")
    if args.record_floors:
        record_floors_first_run(run_dir, Path(args.profile))
    if args.assert_floors:
        return assert_floors(run_dir, Path(args.profile))
    return 0


if __name__ == '__main__':
    sys.exit(main())
```
