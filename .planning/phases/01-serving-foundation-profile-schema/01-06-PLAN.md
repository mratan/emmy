def _hardware_id() -> str:
    try:
        return socket.gethostname()
    except Exception:
        return "unknown"


def measure_warm_throughput(
    base_url: str,
    model: str,
    *,
    candidate_id: str = "k0-baseline",
    n_samples: int = 3,
    max_tokens: int = 500,
    prompt: str = "Count to 100.",
    warmup_discard: int = 1,
    timeout_s: float = 60.0,
    now: float | None = None,  # hook for tests
) -> ThroughputMeasurement:
    """Measure warm decode tok/s on the given endpoint.

    1. Runs `warmup_discard` decode requests and throws the result away.
    2. Runs `n_samples` warm decode requests; each sample is
       `completion_tokens / elapsed_wall` where elapsed is `time.perf_counter()` delta.
    3. Runs run_sp_ok + run_tool_call + run_generate canaries and records booleans.

    Identical payload shape to tests/integration/test_boot.py::test_throughput_floor
    (enable_thinking=False, temperature=0.0) — the only differences are max_tokens=500
    (Phase C warm-500) and n_samples>=3 for std computation.
    """
    ts = time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime(now))
    hw = _hardware_id()
    payload = {
        "model": model,
        "messages": [{"role": "user", "content": prompt}],
        "max_tokens": max_tokens,
        "temperature": 0.0,
        "chat_template_kwargs": {"enable_thinking": False},
    }
    try:
        # Warmup
        for _ in range(max(0, warmup_discard)):
            httpx.post(f"{base_url}/v1/chat/completions", json=payload, timeout=timeout_s)

        samples: list[float] = []
        for _ in range(n_samples):
            t0 = time.perf_counter()
            r = httpx.post(f"{base_url}/v1/chat/completions", json=payload, timeout=timeout_s)
            elapsed = max(time.perf_counter() - t0, 1e-6)
            r.raise_for_status()
            body = r.json()
            ct = int((body.get("usage") or {}).get("completion_tokens", 0))
            if ct <= 0:
                raise RuntimeError(f"completion_tokens={ct}; unusable sample")
            samples.append(ct / elapsed)

        # Canaries — full suite per pitfall #5 discipline
        ok_sp, _ = run_sp_ok(base_url, model)
        ok_tc, _ = run_tool_call(base_url, model, load_default_tool_schema())
        ok_gen, _, _ = run_generate(base_url, model)

        return ThroughputMeasurement(
            candidate_id=candidate_id,
            samples_tokps=[round(s, 3) for s in samples],
            mean=round(statistics.fmean(samples), 2),
            std=round(statistics.pstdev(samples), 3) if len(samples) >= 2 else 0.0,
            p50=round(sorted(samples)[len(samples) // 2], 2),
            canary_sp_ok=bool(ok_sp),
            canary_tool_call=bool(ok_tc),
            canary_generate=bool(ok_gen),
            error=None,
            hardware_id=hw,
            ts=ts,
        )
    except Exception as e:
        return ThroughputMeasurement(
            candidate_id=candidate_id,
            samples_tokps=[],
            mean=0.0,
            std=0.0,
            p50=0.0,
            canary_sp_ok=False,
            canary_tool_call=False,
            canary_generate=False,
            error=str(e)[:500],
            hardware_id=hw,
            ts=ts,
        )


def decide_winner(
    measurements: list[ThroughputMeasurement], floor_tokps: float = 60.0
) -> str | None:
    """Return the id of the first candidate whose mean >= floor AND all canaries pass.

    Excludes k0-baseline. If no candidate qualifies, returns None.
    """
    for m in measurements:
        if m.candidate_id == "k0-baseline":
            continue
        if (
            m.error is None
            and m.mean >= floor_tokps
            and m.canary_sp_ok
            and m.canary_tool_call
            and m.canary_generate
        ):
            return m.candidate_id
    return None

def _stop_container(name: str = "emmy-serve") -> None:
    for args in (["docker", "stop", name], ["docker", "rm", name]):
        subprocess.run(args, capture_output=True)


def _boot_with_env(profile: Path, port: int, extra_env: dict[str, str]) -> None:
    """Boot emmy-serve with additional env vars layered on top of serving.yaml.env.

    We invoke scripts/start_emmy.sh and export the candidate env; start_emmy.sh
    must forward these to `docker run -e`. If the current start_emmy.sh does NOT
    propagate unknown envs, this function invokes `python -m emmy_serve.boot.runner
    render-docker-args` directly and adds `-e K=V` pairs per candidate.
    READ start_emmy.sh AT TASK TIME to confirm which path is correct.
    """
    env = os.environ.copy()
    env.update(extra_env)
    r = subprocess.run(
        ["./scripts/start_emmy.sh", "--profile", str(profile), "--port", str(port)],
        env=env,
    )
    if r.returncode != 0:
        raise RuntimeError(f"start_emmy.sh exited {r.returncode}")


def _wait_for_models(base_url: str, timeout_s: float = 300.0) -> None:
    deadline = time.monotonic() + timeout_s
    while time.monotonic() < deadline:
        try:
            r = httpx.get(f"{base_url}/v1/models", timeout=5.0)
            if r.status_code == 200:
                return
        except Exception:
            pass
        time.sleep(2.0)
    raise RuntimeError(f"/v1/models not ready within {timeout_s}s")


def _apply_serving_yaml_patch(profile: Path, patch: dict | None) -> dict | None:
    """Apply a serving.yaml patch in-place; return the original content for restore."""
    if not patch:
        return None
    sp = profile / "serving.yaml"
    original = sp.read_text(encoding="utf-8")
    data = yaml.safe_load(original)
    for section, fields in patch.items():
        data.setdefault(section, {}).update(fields)
    write_text_atomic(sp, yaml.safe_dump(data, sort_keys=False))
    # Rehash so the profile validates during the candidate's boot.
    subprocess.run(["uv", "run", "emmy", "profile", "hash", str(profile), "--write"], check=False)
    return {"path": str(sp), "original": original}


def _restore_serving_yaml(state: dict | None) -> None:
    if not state:
        return
    Path(state["path"]).write_text(state["original"], encoding="utf-8")
    subprocess.run(
        ["uv", "run", "emmy", "profile", "hash", str(Path(state["path"]).parent), "--write"],
        check=False,
    )

        "max_tokens": args.max_tokens,

        "winner_id": winner,
        "decision": decision,

            {

                "samples_tokps": m.samples_tokps,

                "error": m.error,
                "hardware_id": m.hardware_id,

    }
    write_json_atomic(layout.summary_path.with_name("results.json"), results)
