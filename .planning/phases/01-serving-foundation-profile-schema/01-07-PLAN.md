the field as missing. All thermal floor math (compute_floors in replay.py) already tolerates
missing gpu_clock_mhz keys via `if "gpu_clock_mhz" in s`, so the fix is self-contained in
the sampler's parse step.

```python
# Drop-in replacement for _sample. Signature & return-type unchanged: dict | None.

_NA_SENTINELS = {"[N/A]", "N/A", "", "nan"}


def _parse_float_or_none(s: str) -> float | None:
    """Return float(s) or None if s is an nvidia-smi sentinel for unsupported."""
    s2 = (s or "").strip()
    if s2.casefold() in {x.casefold() for x in _NA_SENTINELS}:
        return None
    try:
        return float(s2)
    except ValueError:
        return None


@staticmethod
def _sample() -> dict | None:
    """Return one nvidia-smi sample or None if structurally unavailable.

    Structurally unavailable = subprocess failure, empty output, or fewer than
    5 CSV fields. Well-formed rows with `[N/A]` or `N/A` in individual numeric
    fields keep the other fields; the bad key is OMITTED from the returned
    dict (matches compute_floors' `if "gpu_clock_mhz" in s` contract).
    """
    try:
        out = subprocess.check_output(
            [
                "nvidia-smi",
                "--query-gpu=timestamp,utilization.gpu,"
                "clocks.current.graphics,temperature.gpu,memory.used",
                "--format=csv,noheader,nounits",
            ],
            timeout=5,
            text=True,
            stderr=subprocess.DEVNULL,
        )
    except Exception:
        return None
    lines = [line.strip() for line in out.strip().splitlines() if line.strip()]
    if not lines:
        return None
    parts = [p.strip() for p in lines[0].split(",")]
    if len(parts) < 5:
        return None
    ts, util, clock, temp, mem = parts
    sample: dict = {"ts": ts}
    for key, raw in (
        ("gpu_util_pct", util),
        ("gpu_clock_mhz", clock),
        ("gpu_temp_c", temp),
        ("memory_used_mb", mem),
    ):
        v = _parse_float_or_none(raw)
        if v is not None:
            sample[key] = v
    return sample
```
