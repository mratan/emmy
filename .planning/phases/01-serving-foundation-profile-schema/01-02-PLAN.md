def append_jsonl_atomic(path: Path, obj: dict) -> None:
    """Append one JSON line, fsync'd. For event streams (kv-finder iterations, thermal samples, canary log)."""
    import json, os
    path.parent.mkdir(parents=True, exist_ok=True)
    line = json.dumps(obj, sort_keys=True, separators=(',', ':')) + '\n'
    with open(path, 'a', encoding='utf-8') as f:
        f.write(line)
        f.flush()
        os.fsync(f.fileno())

**5. `emmy_serve/profile/loader.py`** — file-I/O boundary:

```python
from __future__ import annotations
import yaml
from pathlib import Path
from dataclasses import dataclass
from pydantic import ValidationError
from .schema import ServingConfig, HarnessConfig, ProfileYaml, ProfileConfigError
from .hasher import hash_bundle


@dataclass(frozen=True)
class ProfileRef:
    id: str
    version: str
    hash: str
    path: Path


def _load_yaml(path: Path) -> dict:
    if not path.exists():
        raise ProfileConfigError(f"file not found: {path}")
    try:
        return yaml.safe_load(path.read_text(encoding='utf-8')) or {}
    except yaml.YAMLError as e:
        raise ProfileConfigError(f"yaml parse error in {path}: {e}") from e


def _dotted_path(err_loc: tuple) -> str:
    return '.'.join(str(p) for p in err_loc)


def load_serving(path: Path) -> ServingConfig:
    raw = _load_yaml(path)
    try:
        return ServingConfig(**raw)
    except ValidationError as e:
        errs = '\n  '.join(f"{_dotted_path(err['loc'])}: {err['msg']}" for err in e.errors())
        raise ProfileConfigError(f"serving.yaml invalid ({path}):\n  {errs}") from e


def load_harness(path: Path) -> HarnessConfig:
    raw = _load_yaml(path)
    try:
        return HarnessConfig(**raw)
    except ValidationError as e:
        errs = '\n  '.join(f"{_dotted_path(err['loc'])}: {err['msg']}" for err in e.errors())
        raise ProfileConfigError(f"harness.yaml invalid ({path}):\n  {errs}") from e


def load_profile_manifest(path: Path) -> ProfileYaml:
    raw = _load_yaml(path)
    try:
        return ProfileYaml(**raw)
    except ValidationError as e:
        errs = '\n  '.join(f"{_dotted_path(err['loc'])}: {err['msg']}" for err in e.errors())
        raise ProfileConfigError(f"profile.yaml invalid ({path}):\n  {errs}") from e


def load_profile(bundle_dir: Path) -> tuple[ServingConfig, HarnessConfig, ProfileRef]:
    serving = load_serving(bundle_dir / "serving.yaml")
    harness = load_harness(bundle_dir / "harness.yaml")
    manifest = load_profile_manifest(bundle_dir / "profile.yaml")
    return serving, harness, ProfileRef(
        id=manifest.profile.id,
        version=manifest.profile.version,
        hash=manifest.profile.hash,
        path=bundle_dir,
    )
```

```python
from __future__ import annotations
import sys
from pathlib import Path
from .loader import load_serving, load_harness, load_profile_manifest, ProfileConfigError
from .hasher import hash_bundle, HasherError
from .schema import ProfileConfigError as _SchemaErr  # if distinct; else reuse

EXIT_OK = 0
EXIT_SCHEMA = 1
EXIT_HASH_MISMATCH = 2
EXIT_CANONICALIZATION = 3
EXIT_POLICY = 4


def validate_bundle(bundle_dir: Path, *, fix_hash: bool = False, strict: bool = True) -> int:
    """Run all 3 layers of validation; return exit code per §5 spec."""
    try:
        # 1. Schema layer
        try:
            serving = load_serving(bundle_dir / "serving.yaml")
            harness = load_harness(bundle_dir / "harness.yaml")
            manifest = load_profile_manifest(bundle_dir / "profile.yaml")
        except ProfileConfigError as e:
            msg = str(e)
            # cross-field policy failures raise with "must equal" in the message
            if "must equal" in msg:
                print(f"ERROR (policy): {msg}", file=sys.stderr)
                return EXIT_POLICY
            print(f"ERROR (schema): {msg}", file=sys.stderr)
            return EXIT_SCHEMA

        # 2. Canonicalization + hash layer
        try:
            computed = hash_bundle(bundle_dir)
        except HasherError as e:
            print(f"ERROR (canonicalization): {e}", file=sys.stderr)
            return EXIT_CANONICALIZATION

        stored = manifest.profile.hash
        if computed != stored:
            if fix_hash:
                _rewrite_hash(bundle_dir / "profile.yaml", computed)
                print(f"hash rewritten: {stored} -> {computed}")
                return EXIT_OK
            print(
                f"ERROR: profile hash mismatch for {bundle_dir}/\n"
                f"  stored (profile.yaml):  {stored}\n"
                f"  computed (just now):    {computed}\n"
                f"Any edit to v1 after creation is disallowed (D-03, PROFILE-06).\n"
                f"To change this profile, create profiles/{manifest.profile.id}/v2/ with your edits.",
                file=sys.stderr,
            )
            return EXIT_HASH_MISMATCH
        return EXIT_OK

    except Exception as e:  # noqa
        print(f"ERROR (unexpected): {e}", file=sys.stderr)
        return EXIT_SCHEMA


def _rewrite_hash(profile_yaml_path: Path, new_hash: str) -> None:
    """Rewrite profile.yaml.hash in place, preserving ordering."""
    import yaml
    raw = yaml.safe_load(profile_yaml_path.read_text(encoding='utf-8'))
    raw['profile']['hash'] = new_hash
    # atomic write to preserve durability
    from ..diagnostics.atomic import write_text_atomic
    write_text_atomic(profile_yaml_path, yaml.safe_dump(raw, sort_keys=False))
```

from __future__ import annotations
import argparse, sys
from pathlib import Path
from .profile.immutability import validate_bundle, _rewrite_hash
from .profile.hasher import hash_bundle


def _cmd_profile_validate(args) -> int:
    return validate_bundle(Path(args.path), fix_hash=args.fix_hash, strict=args.strict)


def _cmd_profile_hash(args) -> int:
    computed = hash_bundle(Path(args.path))
    if args.write:
        _rewrite_hash(Path(args.path) / "profile.yaml", computed)
        print(computed)
        return 0
    # default: --check
    from .profile.loader import load_profile_manifest
    stored = load_profile_manifest(Path(args.path) / "profile.yaml").profile.hash
    if computed == stored:
        print(computed)
        return 0
    print(f"MISMATCH\n  stored:   {stored}\n  computed: {computed}", file=sys.stderr)
    return 1


def build_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(prog='emmy')
    sub = p.add_subparsers(dest='cmd', required=True)
    profile = sub.add_parser('profile', help='profile bundle management')
    psub = profile.add_subparsers(dest='profile_cmd', required=True)

    v = psub.add_parser('validate', help='run all 3 validation layers')
    v.add_argument('path')
    v.add_argument('--fix-hash', action='store_true', help='rewrite profile.yaml.hash to computed (dev convenience)')
    v.add_argument('--strict', action='store_true', default=True)
    v.set_defaults(_handler=_cmd_profile_validate)

    h = psub.add_parser('hash', help='compute or check bundle hash')
    h.add_argument('path')
    h.add_argument('--write', action='store_true', help='rewrite profile.yaml.hash')
    h.add_argument('--check', action='store_true', default=True)
    h.set_defaults(_handler=_cmd_profile_hash)

    return p


def main(argv: list[str] | None = None) -> int:
    parser = build_parser()
    args = parser.parse_args(argv)
    return args._handler(args)


if __name__ == '__main__':
    sys.exit(main())
